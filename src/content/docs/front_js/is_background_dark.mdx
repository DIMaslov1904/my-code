---
title: Получаем цвет фона (темный или светлый)
---

`isBackgroundDark(color)` -
Передаем цвет - возращает true если он темный и false если он светлый (Нужен hexToRgb.ts либо встроить его)

```ts
import { hexToRgb } from "./hexToRgb";

export function isBackgroundDark(color: string) {
  const rgb = color.includes("#")
    ? hexToRgb(color).match(/\d+/g)
    : color.match(/\d+/g);
  const luma = 186;
  if (!rgb) return true;
  return +rgb[0] * 0.299 + +rgb[1] * 0.587 + +rgb[2] * 0.114 < luma;
}
```
